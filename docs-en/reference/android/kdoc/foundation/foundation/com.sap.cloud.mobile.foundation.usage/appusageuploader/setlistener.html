static fun setListener(@Nullable listener: AppUsageUploader.UploadListener?): Unit
Deprecated: As of release 3.0, replaced by #addUploadListener(UploadListener) to register a usage listener and #removeUploadListener(UploadListener) to unregister a usage listener.
